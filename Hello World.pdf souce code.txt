package es.iessoterohernandez.daw.endes.HelloWorldPdf;
import com.itextpdf.text.Document;
import com.itextpdf.text.DocumentException;
import com.itextpdf.text.Paragraph;
import com.itextpdf.text.pdf.PdfWriter;

import java.io.FileNotFoundException;
import java.io.FileOutputStream;

public class App {
	public static void main(String[] args) {
    // Crear un objeto Document (el archivo PDF)
    Document document = new Document();

    try {
        // Crear un PdfWriter para escribir en el archivo PDF
        PdfWriter.getInstance(document, new FileOutputStream("HelloWorld.pdf"));
        
        // Abrir el documento
        document.open();

        // Agregar contenido al documento (un p√°rrafo con el texto "Hello World")
        document.add(new Paragraph("Hello World!"));

    } catch (DocumentException | FileNotFoundException e) {
        e.printStackTrace();
    } finally {
        // Cerrar el documento
        document.close();
    }

    System.out.println("PDF generado correctamente.");
}
}
